model Limiter 
  extends SI4SO;
  
  parameter Real Limitf=3.74;
  parameter Real Tdf=20;
  parameter Real Taf=120e6;
  parameter Real Kf=20;
  parameter Real Limits=1.05;
  parameter Real Tds=20;
  parameter Real Tas=120e6;
  parameter Real Ks=2;
  
  ObjectStab.Base.Time counterf(start=1e6, fixed=true);
  ObjectStab.Base.Time counters(start=1e6, fixed=true);
protected 
  Real fcorr(start=1e3);
  Real scorr(start=1e3);
  
public 
  ModelicaAdditions.PetriNets.Transition Tr1f;
  ModelicaAdditions.PetriNets.Place12 countf;
  ModelicaAdditions.PetriNets.Transition Tr2f;
  ModelicaAdditions.PetriNets.Place21 waitf(initialState=true);
  ModelicaAdditions.PetriNets.Place11 activatef;
  ModelicaAdditions.PetriNets.Transition Tr3f;
  ModelicaAdditions.PetriNets.Transition Tr4f;
  ModelicaAdditions.PetriNets.Transition Tr1s;
  ModelicaAdditions.PetriNets.Place12 counts;
  ModelicaAdditions.PetriNets.Transition Tr2s;
  ModelicaAdditions.PetriNets.Place21 waits(initialState=true);
  ModelicaAdditions.PetriNets.Place11 activates;
  ModelicaAdditions.PetriNets.Transition Tr3s;
  ModelicaAdditions.PetriNets.Transition Tr4s;
equation 
  connect(countf.inTransition, Tr1f.outTransition);
  connect(Tr1f.inTransition, waitf.outTransition);
  connect(Tr2f.inTransition, countf.outTransition1);
  connect(activatef.inTransition, Tr2f.outTransition);
  connect(activatef.outTransition, Tr3f.inTransition);
  connect(Tr3f.outTransition, waitf.inTransition1);
  connect(Tr4f.inTransition, countf.outTransition2);
  connect(Tr4f.outTransition, waitf.inTransition2);
  connect(counts.inTransition, Tr1s.outTransition);
  connect(Tr1s.inTransition, waits.outTransition);
  connect(Tr2s.inTransition, counts.outTransition1);
  connect(activates.inTransition, Tr2s.outTransition);
  connect(activates.outTransition, Tr3s.inTransition);
  connect(Tr3s.outTransition, waits.inTransition1);
  connect(Tr4s.inTransition, counts.outTransition2);
  connect(Tr4s.outTransition, waits.inTransition2);
  // field current limiter
  Tr1f.condition = u1 > Limitf;
  Tr2f.condition = (time - counterf) > Tdf;
  Tr3f.condition = (time - counterf) > Taf and (u1 < Limitf);
  Tr4f.condition = (u1 < Limitf);
  
  when countf.state then
    counterf = time;
  end when;
  
  fcorr = if (u4 > 0) and (activatef.state) then Kf*(Limitf - u1) else 1e3;
  
  // stator current limiter
  Tr1s.condition = u2 > Limits;
  Tr2s.condition = (time - counters) > Tds;
  Tr3s.condition = (time - counters) > Tas and (u2 < Limits);
  Tr4s.condition = (u2 < Limits);
  
  when counts.state then
    counters = time;
  end when;
  
  scorr = if (u4 > 0) and (activates.state) then Ks*(Limits - u2) else 1e3;
  
  // output equation
  y = noEvent(min(scorr, min(fcorr, u3)));
  
initial equation 
  waitf.state = true;
  countf.state = false;
  activatef.state = false;
  waits.state = true;
  counts.state = false;
  activates.state = false;
end Limiter;

Nordic32 Static Load Model

Information

 
Static load model used in the CIGRE report.
Active power has constant current and Reactive power has 
constant impedance characteristic.

model SweLoad "Nordic32 Static Load Model\t" 
  extends ObjectStab.Loads.Partials.Load;
  
  outer Base.AngularVelocity wref;
  parameter Real a=0.75;
equation 
  Pl = P0*(V/V0)*wref^a;
  Ql = Q0*(V/V0)^2*wref^a;
end SweLoad;
