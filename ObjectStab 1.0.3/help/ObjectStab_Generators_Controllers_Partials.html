model ExcLimiter "Limiter Logic For Excitation System" 
  extends Modelica.Blocks.Interfaces.SISO;
  parameter Real DelayTime=10;
  parameter Real Limit=2.1;
  
  discrete Real countstart(start=1e6, fixed=true);
  //  Integer status1;
  //  Integer status2;
  ModelicaAdditions.PetriNets.Transition LimitExc;
  ModelicaAdditions.PetriNets.Transition LimitOK;
  ModelicaAdditions.PetriNets.Transition TimerExp;
  ModelicaAdditions.PetriNets.Place11 activated;
  ModelicaAdditions.PetriNets.Transition UnActivate;
  ModelicaAdditions.PetriNets.Place21 wait(initialState=true);
  ModelicaAdditions.PetriNets.Place12 count;
equation 
  connect(LimitExc.inTransition, wait.outTransition);
  connect(LimitExc.outTransition, count.inTransition);
  connect(TimerExp.outTransition, activated.inTransition);
  connect(count.outTransition1, LimitOK.inTransition);
  connect(activated.outTransition, UnActivate.inTransition);
  connect(UnActivate.outTransition, wait.inTransition2);
  connect(LimitOK.outTransition, wait.inTransition1);
  connect(count.outTransition2, TimerExp.inTransition);
  LimitExc.condition = y > Limit;
  LimitOK.condition = y < (Limit - 1e-5);
  UnActivate.condition = y < (Limit - 1e-5);
  TimerExp.condition = ((time - countstart) > DelayTime) and (y > Limit);
  
  when count.state then
    countstart = time;
    //   status1 = LogVariable(count.state);
  end when;
  when activated.state then
    //  status2 = LogVariable(activated.state);
  end when;
  
  y = if activated.state then min(u, Limit) else u;
initial equation 
  count.state = false;
  wait.state = true;
  activated.state = false;
end ExcLimiter;
